CREATE TABLE rec (
    id INT AUTO_INCREMENT,
    artist VARCHAR(255),
    size VARCHAR(255),
    label VARCHAR(255),
    title VARCHAR(255),
	  kan VARCHAR(255),
    lev VARCHAR(255),
    price DOUBLE,
    discogs VARCHAR(255),
    genre VARCHAR(255),
    PRIMARY key (id)
);

CREATE TABLE recordstoreusers (
  id INT AUTO_INCREMENT,
  email VARCHAR(100) NOT NULL,
  user_password VARCHAR(255) NOT NULL,
  user_role VARCHAR(50) NOT NULL,
  PRIMARY KEY (email),
  UNIQUE (id)
);

CREATE TABLE shoppingcart (
  user_id INT,
  record_id INT,
  PRIMARY KEY (user_id, record_id),
  FOREIGN KEY (user_id) REFERENCES recordstoreusers (id),
  FOREIGN KEY (record_id) REFERENCES rec (id)
);

CREATE TABLE shoppingcart (
  user_id INT,
  record_id INT,
  PRIMARY KEY (record_id),
  FOREIGN KEY (user_id) REFERENCES recordstoreusers (id) ON DELETE CASCADE,
  FOREIGN KEY (record_id) REFERENCES rec (id) ON DELETE CASCADE
);

CREATE TABLE conversations (
    id INT AUTO_INCREMENT PRIMARY KEY,
    user1_id INT NOT NULL,
    user2_id INT NOT NULL,
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    FOREIGN KEY (user1_id) REFERENCES recordstoreusers(id),
    FOREIGN KEY (user2_id) REFERENCES recordstoreusers(id)
);

CREATE TABLE messages (
    id INT AUTO_INCREMENT PRIMARY KEY,
    conversation_id INT NOT NULL,
    sender_id INT NOT NULL,
    message TEXT NOT NULL,
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    FOREIGN KEY (conversation_id) REFERENCES conversations(id),
    FOREIGN KEY (sender_id) REFERENCES recordstoreusers(id)
);

CREATE TABLE orders (
  id INT AUTO_INCREMENT PRIMARY KEY,
  user_id INT,
  customer_name VARCHAR(255),
  customer_phone VARCHAR(255),
  customer_email VARCHAR(255),
  customer_paymentoption VARCHAR(255),
  customer_shippingoption VARCHAR(255),
  customer_address VARCHAR(255),
  FOREIGN KEY (user_id) REFERENCES recordstoreusers (id) ON DELETE CASCADE
);

CREATE TABLE order_items (
  order_id INT,
  record_id INT,
  FOREIGN KEY (order_id) REFERENCES orders (id) ON DELETE CASCADE,
	FOREIGN KEY (record_id) REFERENCES rec (id) ON DELETE CASCADE
);

DROP TABLE shoppingcart;
